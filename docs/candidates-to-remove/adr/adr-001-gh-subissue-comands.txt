# GitHub CLI issue creation commands for ADR-001 follow-on sub-issues
# File name intentionally matches requested spelling: adr-001-gh-subissue-comands.txt
#
# This script creates ADR-001 follow-on issues and links EACH one as a sub-issue
# under parent issue #1060.
#
# Usage:
# 1) Ensure GitHub CLI auth: gh auth status
# 2) Run from repo root:
#    bash docs/adr/adr-001-gh-subissue-comands.txt

set -euo pipefail

# Optional label bootstrap (safe if label already exists)

link_subissue() {
  local child_issue_url="$1"

  gh api graphql \
    -f query='mutation($issueId: ID!, $subIssueUrl: String!) { addSubIssue(input: { issueId: $issueId, subIssueUrl: $subIssueUrl, replaceParent: false }) { issue { number url } subIssue { number url } } }' \
    -f issueId="$PARENT_ISSUE_NODE_ID" \
    -f subIssueUrl="$child_issue_url" \
    >/dev/null
}

create_and_link() {
  local title="$1"
  local milestone="$2"
  local labels_csv="$3"
  local body_file="$4"

  local cmd=(gh issue create --repo "$REPO" --title "$title" --milestone "$milestone" --body-file "$body_file")

  IFS=',' read -r -a labels <<< "$labels_csv"
  for label in "${labels[@]}"; do
    cmd+=(--label "$label")
  done

  local child_url
  child_url=$("${cmd[@]}")

  if [[ -z "$child_url" ]]; then
    echo "Failed to create issue: $title" >&2
    exit 1
  fi

  link_subissue "$child_url"
  echo "Created and linked sub-issue: $child_url"
}
