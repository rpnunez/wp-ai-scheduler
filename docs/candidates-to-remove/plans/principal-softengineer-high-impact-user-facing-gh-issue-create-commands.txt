# GitHub CLI issue creation commands for the Principal Software Engineer high-impact user-facing plan
# Usage:
# 1) Ensure you are authenticated: gh auth status
# 2) Run from repo root: bash docs/plans/principal-softengineer-high-impact-user-facing-gh-issue-create-commands.txt
# 3) These commands target: rpnunez/wp-ai-scheduler

set -euo pipefail

REPO="rpnunez/wp-ai-scheduler"

gh label create "user-facing" --repo "$REPO" --color "5319E7" --description "Direct user-facing functionality" >/dev/null 2>&1 || true
